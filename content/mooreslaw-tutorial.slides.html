Model¶

Moores_law(year) = (e^B_M) × (e^(A_M × year))

A_M = (log 2) ÷ 2

B_M = (log 2250) - A_M × 1971

In [2]:
     #        exp←*
     #        log←*○
     A_M = np.log(2) / 2
run("A_M ←     (⍟ 2) ÷ 2")

     B_M = np.log(2250) - A_M * 1971
run("B_M ←     (⍟ 2250) - A_M × 1971")

     Moores_law = lambda year: np.exp(B_M) * np.exp(A_M * year)
run("Moores_law ← {      year←⍵ ⋄  (* B_M) ×      *(A_M × year)} ⋄")
